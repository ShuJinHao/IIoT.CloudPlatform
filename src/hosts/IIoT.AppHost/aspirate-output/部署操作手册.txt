## 一、本地构建推送（每次代码更新后执行）
==================================================

在 aspirate-output 目录下执行：

powershell -ExecutionPolicy Bypass -File .\build-push.ps1

# 如果只改了配置文件没改代码，跳过构建：
powershell -ExecutionPolicy Bypass -File .\build-push.ps1 -SkipBuild


==================================================
## 二、服务器部署（通过 MobaXterm 执行）
==================================================

### 情况A：代码有更新（构建推送完之后执行）
cd /opt/iiot-app
export $(grep -v '^#' .env | xargs)
docker stack deploy -c docker-compose.yaml iiot-app --with-registry-auth

### 情况B：只更新了配置文件（.env / docker-compose.yaml / nginx.conf）
# 1. 先通过 MobaXterm SFTP 把文件拖拽上传到 /opt/iiot-app/
# 2. 再执行：
cd /opt/iiot-app
export $(grep -v '^#' .env | xargs)
docker stack deploy -c docker-compose.yaml iiot-app --with-registry-auth


==================================================
## 三、验证部署结果
==================================================

# 查看所有服务状态
docker stack ps iiot-app

# 正常状态应显示：
# iiot-httpapi  × 3  Running
# iiot-web      × 2  Running
# nginx-gateway × 1  Running
# postgres      × 1  Running
# redis-cache   × 1  Running
# aspire-dashboard × 1  Running

# 查看服务日志（替换 服务名）
docker service logs iiot-app_iiot-httpapi --tail 50

## 五、常用运维命令
==================================================

# 查看服务列表
docker stack services iiot-app

# 强制重启某个服务（不重新部署）
docker service update --force iiot-app_iiot-httpapi

# 扩缩容
docker service scale iiot-app_iiot-httpapi=3

# 移除整个 Stack（慎用）
docker stack rm iiot-app

# 查看容器日志
docker service logs iiot-app_iiot-httpapi -f
